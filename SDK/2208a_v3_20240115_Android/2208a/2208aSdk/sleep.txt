    public static int getSleepLevelerOneMinute(float data) {
        int level = 2;
        if (data == 1f) {//深睡Deep sleep
            level = 0;
        } else if (data == 2f) {//浅睡Light sleep
            level = 1;
        } else if (data == 3f) {//rem
            level = 3;
        }else{//清醒Awake
            level = 2;
        }
        return level;
    }

 public static int getSleepLeveler(float data) {
        int level = 2;
        if (data >= 0 && data <= 2f) {//深睡Deep sleep
            level = 0;
        } else if (data > 2f && data <= 8f) {//浅睡 Light sleep
            level = 1;
        } else if (data > 8f&& data <= 20f) {//rem
            level = 3;
        }else{//清醒Awake
            level = 2;
        }
        return level;
    }
